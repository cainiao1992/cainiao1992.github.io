network:
  version: 2
  ethernets:
    eth0:
      dhcp4: true
      dhcp6: false
      addresses:
        - XXXX:XXXX:XXXX:XXXX:XXXX:XXXX:XXXX:XXXX/48
      nameservers:
        addresses:
          - 223.5.5.5
          - 223.6.6.6
          -  2400:3200::1
          - 2400:3200:baba::1
      routes:
        - to: default
          via: "XXXX:XXXX:XXXX::1"
          on-link: true

sudo netplan apply
